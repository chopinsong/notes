2.2 for 循环

定义： 用于遍历序列（列表、元组、字符串等）的循环结构。

描述： 常用于迭代已知次数的循环。

# 基本 for 循环

2.3 while 循环

定义： 在条件为真时重复执行代码块的循环结构。

描述： 适用于不知道确切循环次数的情况。

# 基本 while 循环

while count < 5:
    print(count)
    count += 1

# 使用 while 循环进行用户输入

    user_input = input("请输入 'quit' 退出: ")
    if user_input.lower() == 'quit':
        break
    print(f"您输入了: {user_input}")

# 计算阶乘
n = 5
factorial = 1
while n > 0:
    factorial *= n
    n -= 1
print(f"5! = {factorial}")

2.4 循环控制语句

定义： 用于控制循环执行流程的语句。

描述： 包括 break、continue 和 else 子句。

# break 语句 - 终止循环
for i in range(10):
    if i == 5:
        break
    print(i)  # 输出 0, 1, 2, 3, 4

# continue 语句 - 跳过当前迭代
for i in range(10):
    if i % 2 == 0:
        continue
    print(i)  # 输出 1, 3, 5, 7, 9

# else 子句 - 循环正常结束时执行
for i in range(3):
    print(i)
else:
    print("循环正常结束")

# 在有 break 的情况下，else 不会执行
for i in range(10):
    if i == 5:
        break
    print(i)
else:
    print("这不会被打印")

3. 函数

3.1 定义函数

定义： 函数是用于执行单一动作的组织、可重用代码块。

描述： 函数有助于将程序分为更小和模块化的块。

# 基本函数定义
def greet(name):
    return "Hello, " + name

# 调用函数
result = greet("Alice")
print(result)  # 输出: Hello, Alice

# 带默认参数的函数
def greet_with_title(name, title="先生"):
    return f"您好，{title} {name}！"

print(greet_with_title("张三"))  # 您好，先生 张三！
print(greet_with_title("李四", "女士"))  # 您好，女士 李四！

# 可变参数函数
def sum_numbers(*args):
    return sum(args)

print(sum_numbers(1, 2, 3, 4, 5))  # 15

# 关键字参数函数
def create_profile(**kwargs):
    return kwargs

profile = create_profile(name="张三", age=25, city="北京")
print(profile)  # {'name': '张三', 'age': 25, 'city': '北京'}

# 混合参数函数
def complex_function(required, default="默认值", *args, **kwargs):
    print(f"必需参数: {required}")
    print(f"默认参数: {default}")
    print(f"可变参数: {args}")
    print(f"关键字参数: {kwargs}")

complex_function("必需", "自定义", 1, 2, 3, key1="value1", key2="value2")

3.2 函数作用域

定义： 变量的可见性和生命周期范围。

描述： 理解局部变量和全局变量的区别。

# 全局变量
global_var = "我是全局变量"

def test_scope():
    # 局部变量
    local_var = "我是局部变量"
    print(global_var)  # 可以访问全局变量
    print(local_var)

test_scope()
# print(local_var)  # 错误：无法访问局部变量

# 修改全局变量

    global counter
    counter += 1
    return counter

print(increment())  # 1
print(increment())  # 2

# 嵌套函数和闭包
def outer_function(x):
    def inner_function(y):
        return x + y
    return inner_function

add_10 = outer_function(10)
print(add_10(5))  # 15

3.3 Lambda 函数

定义： 小型匿名函数，可以有任意数量的输入参数，但只能有一个表达式。

描述： 对于简短和简单的函数有用，通常用于函数式编程结构中。

# 基本 Lambda 函数
square = lambda x: x * x
print(square(5))  # 25

# 多参数 Lambda
add = lambda x, y: x + y
print(add(3, 4))  # 7

# 在内置函数中使用 Lambda
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x**2, numbers))
print(squared)  # [1, 4, 9, 16, 25]

# 过滤偶数
even_numbers = list(filter(lambda x: x % 2 == 0, numbers))
print(even_numbers)  # [2, 4]

# 排序
students = [('Alice', 85), ('Bob', 90), ('Charlie', 78)]
students.sort(key=lambda x: x[1])  # 按分数排序
print(students)  # [('Charlie', 78), ('Alice', 85), ('Bob', 90)]

3.4 高阶函数

定义： 接受函数作为参数或返回函数的函数。

描述： 函数式编程的重要概念，使代码更加灵活和模块化。

# 函数作为参数
def apply_operation(x, y, operation):
    return operation(x, y)

def multiply(a, b):
    return a * b

result = apply_operation(5, 3, multiply)
print(result)  # 15

# 使用 Lambda 作为参数
result2 = apply_operation(5, 3, lambda a, b: a + b)
print(result2)  # 8

# 返回函数的函数
def create_multiplier(factor):
    def multiplier(x):
        return x * factor
    return multiplier

double = create_multiplier(2)
triple = create_multiplier(3)

print(double(5))  # 10
print(triple(5))  # 15

# 装饰器示例
def timer_decorator(func):
    import time
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} 执行时间: {end - start:.4f} 秒")
        return result
    return wrapper

@timer_decorator
def slow_function():
    import time
    time.sleep(1)
    return "完成！"

slow_function()  # 会显示执行时间

4. 集合

4.1 列表 (List)

定义： 有序的、可变的项目集合。

描述： 列表可以在单个变量中存储多个项目，允许动态数据处理。

# 创建列表
fruits = ["apple", "banana", "cherry"]
numbers = [1, 2, 3, 4, 5]
mixed = [1, "hello", 3.14, True]

# 访问列表元素

print(fruits[-1])    # cherry (负索引)
print(fruits[1:3])   # ['banana', 'cherry'] (切片)

# 修改列表
fruits[1] = "blueberry"
print(fruits)        # ['apple', 'blueberry', 'cherry']

# 添加元素
fruits.append("orange")        # 末尾添加
fruits.insert(1, "grape")      # 指定位置插入
fruits.extend(["kiwi", "mango"]) # 添加多个元素

# 删除元素
fruits.remove("grape")         # 删除指定值
popped = fruits.pop()          # 删除并返回最后一个元素
popped_index = fruits.pop(1)   # 删除指定索引的元素
del fruits[0]                  # 删除指定索引的元素

# 列表方法
print(len(fruits))             # 获取长度
print(fruits.count("apple"))   # 计算某元素出现次数
print(fruits.index("cherry"))  # 查找元素索引
fruits.sort()                  # 排序
fruits.reverse()               # 反转

# 列表推导式

print(even_squares) # [0, 4, 16, 36, 64]

# 遍历字典

    "student2": {"name": "李四", "grade": 90}

5. 面向对象编程

5.1 类和对象

定义： 类是用于创建对象（特定数据结构）的蓝图。

描述： Python 是一种面向对象的语言，使用类来定义数据类型。

# 基本类定义
class Dog:
    # 类变量
    species = "Canis lupus"
    
    # 构造函数
    def __init__(self, name, age):
        self.name = name      # 实例变量
        self.age = age        # 实例变量
    
    # 实例方法
    def bark(self):
        return f"{self.name} says: Woof!"
    
    def get_info(self):
        return f"Name: {self.name}, Age: {self.age}, Species: {self.species}"
    
    # 类方法
    @classmethod
    def create_puppy(cls, name):
        return cls(name, 0)
    
    # 静态方法
    @staticmethod
    def sound():
        return "Dogs make woof sound"

# 创建对象
my_dog = Dog("Rex", 3)
puppy = Dog.create_puppy("Buddy")

print(my_dog.bark())        # Rex says: Woof!
print(my_dog.get_info())    # Name: Rex, Age: 3, Species: Canis lupus
print(Dog.sound())          # Dogs make woof sound
print(puppy.get_info())     # Name: Buddy, Age: 0, Species: Canis lupus

5.2 继承

定义： 一种机制，新类可以继承现有类的属性和方法。

描述： 支持面向对象编程中的代码重用和扩展。

# 基础类
class Animal:
    def __init__(self, name, species):
        self.name = name
        self.species = species
    
    def speak(self):
        print(f"{self.name} makes a sound")
    
    def get_info(self):
        return f"Name: {self.name}, Species: {self.species}"

# 继承类
class Dog(Animal):
    def __init__(self, name, breed):
        super().__init__(name, "Canis lupus")  # 调用父类构造函数
        self.breed = breed
    
    def speak(self):  # 方法重写
        print(f"{self.name} barks: Woof!")
    
    def fetch(self):  # 新方法
        print(f"{self.name} fetches the ball")

class Cat(Animal):
    def __init__(self, name, indoor=True):
        super().__init__(name, "Felis catus")
        self.indoor = indoor
    
    def speak(self):  # 方法重写
        print(f"{self.name} meows: Meow!")
    
    def purr(self):  # 新方法
        print(f"{self.name} purrs contentedly")

# 使用继承
dog = Dog("Buddy", "Golden Retriever")
cat = Cat("Whiskers", indoor=True)

dog.speak()     # Buddy barks: Woof!
cat.speak()     # Whiskers meows: Meow!
dog.fetch()     # Buddy fetches the ball
cat.purr()      # Whiskers purrs contentedly

# 多态性
animals = [dog, cat]
for animal in animals:
    animal.speak()  # 每个动物都会有不同的叫声

5.3 封装

定义： 将数据和方法封装在类中，控制对数据的访问。

描述： 通过私有变量和方法来隐藏内部实现细节。

# 封装示例
class BankAccount:
    def __init__(self, account_number, initial_balance=0):
        self.account_number = account_number
        self.__balance = initial_balance  # 私有变量
    
    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount
            print(f"存入 {amount} 元，当前余额: {self.__balance} 元")
        else:
            print("存入金额必须大于0")
    
    def withdraw(self, amount):
        if amount > 0 and amount <= self.__balance:
            self.__balance -= amount
            print(f"取出 {amount} 元，当前余额: {self.__balance} 元")
        else:
            print("余额不足或金额无效")
    
    def get_balance(self):
        return self.__balance
    
    def __str__(self):
        return f"账户: {self.account_number}, 余额: {self.__balance} 元"

# 使用封装
account = BankAccount("123456789", 1000)
print(account)              # 账户: 123456789, 余额: 1000 元
account.deposit(500)        # 存入 500 元，当前余额: 1500 元

5.4 多态性

定义： 不同的类可以有相同的方法名，但实现不同的功能。

描述： 允许使用统一的接口处理不同类型的对象。

# 多态性示例
class Shape:
    def area(self):
        pass
    
    def perimeter(self):
        pass

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius
    
    def area(self):
        return 3.14159 * self.radius ** 2
    
    def perimeter(self):
        return 2 * 3.14159 * self.radius

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height
    
    def area(self):
        return self.width * self.height
    
    def perimeter(self):
        return 2 * (self.width + self.height)

class Triangle(Shape):
    def __init__(self, base, height, side1, side2, side3):
        self.base = base
        self.height = height
        self.side1 = side1
        self.side2 = side2
        self.side3 = side3
    
    def area(self):
        return 0.5 * self.base * self.height
    
    def perimeter(self):
        return self.side1 + self.side2 + self.side3

# 多态性使用
shapes = [
    Circle(5),
    Rectangle(4, 6),
    Triangle(3, 4, 3, 4, 5)
]

for shape in shapes:
    print(f"面积: {shape.area():.2f}")
    print(f"周长: {shape.perimeter():.2f}")
    print("---")

5.5 特殊方法（魔术方法）

定义： 以双下划线开头和结尾的方法，定义对象的特殊行为。

描述： 允许自定义类与Python内置操作符和函数的交互。

# 特殊方法示例
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __str__(self):
        return f"Point({self.x}, {self.y})"
    
    def __repr__(self):
        return f"Point({self.x}, {self.y})"
    
    def __add__(self, other):
        return Point(self.x + other.x, self.y + other.y)
    
    def __sub__(self, other):
        return Point(self.x - other.x, self.y - other.y)
    
    def __eq__(self, other):
        return self.x == other.x and self.y == other.y
    
    def __lt__(self, other):
        return (self.x**2 + self.y**2) < (other.x**2 + other.y**2)
    
    def __len__(self):
        return int((self.x**2 + self.y**2)**0.5)
    
    def __getitem__(self, key):
        if key == 0:
            return self.x
        elif key == 1:
            return self.y
        else:
            raise IndexError("Index out of range")

# 使用特殊方法
p1 = Point(3, 4)
p2 = Point(1, 2)

print(p1)           # Point(3, 4)
print(p1 + p2)      # Point(4, 6)
print(p1 - p2)      # Point(2, 2)
print(p1 == p2)     # False
print(p1 < p2)      # False
print(len(p1))      # 5
print(p1[0])        # 3
print(p1[1])        # 4

6. 错误处理

6.1 Try 和 Except

定义： 用于在 Python 中处理异常（错误）的块。

描述： 通过捕捉和处理异常，帮助防止程序崩溃。

# Try 和 except 示例
try:
    x = 1 / 0
except ZeroDivisionError:
    print("不能除以零")

6.2 Finally

定义： 无论是否发生异常都会执行代码的块。

描述： 用于清理任务，如关闭文件或释放资源。

# Finally 示例
try:
    x = 1 / 0
except ZeroDivisionError:
    print("不能除以零")
finally:
    print("执行完毕")
